CREATE OR REPLACE TRIGGER  "KURS_triger" 
BEFORE INSERT ON KURS
FOR EACH ROW

BEGIN
  SELECT KURS_SEQUENCE.NEXTVAL
  INTO   :new."br_kursa"
  FROM   dual;
END;
/
ALTER TRIGGER  "KURS_triger" ENABLE
/
